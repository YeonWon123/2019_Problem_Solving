위 사각형
가로 N - b/2 ~ n + b/2    세로 0 ~ a - 1

왼쪽 사각형
가로 0 ~ b - 1 		세로 a ~ 2a - 1

오른쪽 사각형
가로 b ~ 2b - 1		세로 a ~ 2a - 1

void divide(가로 시작, 가로 끝, 세로 시작, 세로 끝)
{
	if (세로 끝값 - 세로 시작값 == 3) // 높이가 3일 경우
		// 삼각형을 그린다. 단 저 사각형 좌표에 맞게
		printstar(가로 시작, 가로 끝, 세로 시작, 세로 끝);
	else
	{
		// 재귀 호출
		divide(윗 사각형 범위);
		divide(왼쪽 사각형 범위);
		divide(오른쪽 사각형 범위);
	}

}
